def check_visibility(time = Time.now)
  if (!self.start_date.blank? && !self.end_date.blank?) && (self.end_date.to_time < time || self.start_date.to_time > time || self.remaining_quantity < 1 )
    self.visible = false
    true
  else
    self.visible = true
  end
end

def voucher
  self.line_items.each do |li|
