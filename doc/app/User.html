def self.search(search, user)
  if search
    search_pattern = "%#{search}%"
    where('user_name != ? AND ((name LIKE ?) OR (id like ?) OR (email LIKE ?) OR (wallet > ?)) ', user.user_name, search_pattern, search_pattern, search_pattern, search)
  else
    scoped
  end
end
